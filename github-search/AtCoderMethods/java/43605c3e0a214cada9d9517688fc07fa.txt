public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    double a = scanner.nextDouble(), b = scanner.nextDouble();
    if (a > b) {
        System.out.println("GREATER");
    } else if (a < b) {
        System.out.println("LESS");
    } else {
        System.out.println("EQUAL");
    }
}

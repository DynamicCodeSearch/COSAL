public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int a = sc.nextInt();
    int b = sc.nextInt();
    int c = sc.nextInt();
    int d = sc.nextInt();
    int start = Math.max(a, c);
    int end = Math.min(b, d);
    int ans = end - start;
    if (ans <= 0) {
        System.out.println(0);
    } else {
        System.out.println(ans);
    }
}

public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int a = sc.nextInt();
    int b = sc.nextInt();
    int c = sc.nextInt();
    if (a >= b && a >= c) {
        System.out.println(b + c);
    } else if (b >= c && b >= a) {
        System.out.println(a + c);
    } else if (c >= a && c >= b) {
        System.out.println(a + b);
    }
}

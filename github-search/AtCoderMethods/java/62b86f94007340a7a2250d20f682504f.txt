public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    String s = sc.next();
    int cnt = 0;
    for (int i = 0; i < s.length(); i++) {
        for (int j = 0; j < s.length(); j++) {
            if (i != j && s.charAt(i) == s.charAt(j)) {
                cnt++;
            }
        }
    }
    if (cnt > 0)
        System.out.println("no");
    else
        System.out.println("yes");
}

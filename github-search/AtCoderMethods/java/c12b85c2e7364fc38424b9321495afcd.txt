public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    while (sc.hasNext()) {
        int[] a = new int[3];
        for (int i = 0; i < 3; i++) a[i] = sc.nextInt();
        Arrays.sort(a);
        if (a[0] + a[1] == a[2])
            System.out.println("Yes");
        else
            System.out.println("No");
    }
}

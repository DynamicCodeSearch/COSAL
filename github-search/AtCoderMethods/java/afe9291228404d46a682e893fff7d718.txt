public static void main(String[] args) {
    int A, B, C;
    Scanner scan = new Scanner(System.in);
    A = scan.nextInt();
    B = scan.nextInt();
    C = scan.nextInt();
    if (A == B) {
        System.out.println(C);
    } else if (A == C) {
        System.out.println(B);
    } else if (B == C) {
        System.out.println(A);
    }
}

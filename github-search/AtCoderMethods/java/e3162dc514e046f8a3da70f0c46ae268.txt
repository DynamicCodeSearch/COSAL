public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int W = sc.nextInt();
    int a = sc.nextInt();
    int b = sc.nextInt();
    if (a < b) {
        if (b - (a + W) < 0)
            System.out.println(0);
        else
            System.out.println(b - a - W);
    } else {
        if (a - (b + W) < 0)
            System.out.println(0);
        else
            System.out.println(a - b - W);
    }
}

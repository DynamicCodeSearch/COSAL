public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();
    int k = sc.nextInt();
    int x = sc.nextInt();
    int y = sc.nextInt();
    int ans;
    if ((n - k) >= 0) {
        ans = k * x + (n - k) * y;
    } else {
        ans = n * x;
    }
    System.out.println(ans);
}

public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();
    int[] l = new int[n];
    int[] r = new int[n];
    int sum = 0;
    for (int i = 0; i < n; i++) {
        l[i] = sc.nextInt();
        r[i] = sc.nextInt();
        sum += r[i] - l[i] + 1;
    }
    System.out.println(sum);
}

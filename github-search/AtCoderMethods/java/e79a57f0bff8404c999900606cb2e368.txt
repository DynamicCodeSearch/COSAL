public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    String s = sc.next();
    int counter = 0;
    for (int i = 0; i <= s.length() - 2; i++) {
        if (s.charAt(i) != s.charAt(i + 1))
            counter++;
    }
    System.out.println(String.valueOf(counter));
}

public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    long a = sc.nextLong();
    long b = sc.nextLong();
    long x = sc.nextLong();
    long n = 0;
    if (x == 0) {
        System.out.println(n);
        return;
    }
    n = b / x - a / x;
    if (a % x == 0) {
        n++;
    }
    
    System.out.println(n);
}

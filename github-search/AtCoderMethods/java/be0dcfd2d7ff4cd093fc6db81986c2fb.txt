public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();
    long k = sc.nextLong();
    long[] num = new long[100001];
    long sum = 0;
    for (int i = 0; i < n; i++) {
        int a = sc.nextInt();
        int b = sc.nextInt();
        num[a] += (long) b;
    }
    for (int i = 0; i < 100001; i++) {
        sum += num[i];
        if (sum >= k) {
            System.out.println(i);
            return;
        }
    }
}

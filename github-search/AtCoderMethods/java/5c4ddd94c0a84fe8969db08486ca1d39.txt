public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    int n = scanner.nextInt();
    int answer = 0;
    for (int i = 0; i < n; i++) {
        int l = scanner.nextInt();
        int r = scanner.nextInt();
        answer += r - l + 1;
    }
    System.out.println(answer);
    scanner.close();
}

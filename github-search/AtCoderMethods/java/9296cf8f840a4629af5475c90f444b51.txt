public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int a = sc.nextInt();
    int b = sc.nextInt();
    int c = sc.nextInt();
    if (a == b) {
        System.out.print(c);
    } else if (a == c) {
        System.out.print(b);
    } else {
        System.out.print(a);
    }
}

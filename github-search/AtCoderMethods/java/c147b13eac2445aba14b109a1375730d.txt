public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int N = sc.nextInt();
    int K = sc.nextInt();
    int[] a = new int[N];
    for (int i = 0; i < N; i++) {
        a[i] = sc.nextInt();
    }
    long sum = 0;
    for (int i = 0; i <= N - K; i++) {
        for (int j = i; j < i + K; j++) {
            sum += a[j];
        }
    }
    System.out.println(sum);
}

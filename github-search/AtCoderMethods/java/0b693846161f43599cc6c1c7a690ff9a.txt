public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();
    int[] t = new int[n];
    int min = 100;
    for (int i = 0; i < n; i++) {
        t[i] = sc.nextInt();
        if (t[i] < min) {
            min = t[i];
        }
    }
    System.out.println(min);
}

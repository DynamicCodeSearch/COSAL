public static void main(String[] args) {
    Scanner scan = new Scanner(System.in);
    int N = scan.nextInt();
    int K = scan.nextInt();
    int X = scan.nextInt();
    int Y = scan.nextInt();
    int Hi = N - K;
    int Yen;
    if (Hi > 0) {
        Yen = K * X;
        Yen = Yen + (Hi * Y);
        System.out.println(Yen);
    } else {
        Yen = N * X;
        System.out.println(Yen);
    }
    scan.close();
}

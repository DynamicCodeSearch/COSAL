public static void main(String[] args) {
    
    Scanner sc = new Scanner(System.in);
    String a = sc.next();
    if (a.equals("A")) {
        System.out.println(1);
    } else if (a.equals("B")) {
        System.out.println(2);
    } else if (a.equals("C")) {
        System.out.println(3);
    } else if (a.equals("D")) {
        System.out.println(4);
    } else if (a.equals("E")) {
        System.out.println(5);
    }
}

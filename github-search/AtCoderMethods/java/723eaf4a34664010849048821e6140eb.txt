public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();
    int[] a = new int[n];
    for (int i = 0; i < n; i++) {
        a[i] = sc.nextInt();
    }
    sc.close();
    long count = n;
    int begin = 0;
    for (int i = 1; i < n; i++) {
        if (a[i - 1] < a[i]) {
            count += i - begin;
        } else {
            begin = i;
        }
    }
    System.out.println(count);
}

public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int N = sc.nextInt();
    int K = sc.nextInt();
    int[] L = new int[N];
    for (int i = 0; i < N; i++) {
        L[i] = sc.nextInt();
    }
    Arrays.sort(L);
    int ans = 0;
    for (int i = 1; i <= K; i++) {
        ans += L[N - i];
    }
    System.out.println(ans);
}

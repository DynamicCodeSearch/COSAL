public static void main(String[] args) {
    Scanner in = new Scanner(System.in);
    int n = in.nextInt();
    int x = in.nextInt();
    boolean fOrB = ((double) n / 2.0) >= x;
    if (fOrB) {
        System.out.println(x - 1);
    } else {
        System.out.println(n - x);
    }
    in.close();
}

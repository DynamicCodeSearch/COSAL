public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int a = sc.nextInt();
    int b = sc.nextInt();
    int c = sc.nextInt();
    if (a == b) {
        System.out.println(c);
    }
    if (b == c) {
        System.out.println(a);
    }
    if (c == a) {
        System.out.println(b);
    }
    sc.close();
}

public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int n = sc.nextInt();
    int k = sc.nextInt();
    sc.close();
    int ret = 1;
    for (int i = 0; i < n; i++) {
        if (i == 0)
            ret *= k;
        else
            ret *= k - 1;
    }
    System.out.println(ret);
}

public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int N = sc.nextInt();
    HashSet<Integer> set = new HashSet<Integer>();
    for (int i = 0; i < N; i++) {
        int a = sc.nextInt();
        while (a % 2 == 0) {
            a = a / 2;
        }
        set.add(a);
    }
    System.out.println(set.size());
}

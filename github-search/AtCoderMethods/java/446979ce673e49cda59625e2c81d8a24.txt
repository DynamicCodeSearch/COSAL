public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    String x = sc.nextLine();
    switch(x) {
        case "A":
            System.out.println(1);
            break;
        case "B":
            System.out.println(2);
            break;
        case "C":
            System.out.println(3);
            break;
        case "D":
            System.out.println(4);
            break;
        case "E":
            System.out.println(5);
            break;
        default:
            break;
    }
}

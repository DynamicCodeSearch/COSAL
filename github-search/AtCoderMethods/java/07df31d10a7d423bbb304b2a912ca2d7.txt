public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int a = sc.nextInt();
    int b = sc.nextInt();
    int c = sc.nextInt();
    int sum1 = a + b;
    int sum2 = a + c;
    int sum3 = b + c;
    if (sum1 <= sum2 && sum1 <= sum3) {
        System.out.println(sum1);
    } else if (sum2 <= sum1 && sum2 <= sum3) {
        System.out.println(sum2);
    } else {
        System.out.println(sum3);
    }
}

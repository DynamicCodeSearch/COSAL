public static void main(String[] args) {
    Scanner in = new Scanner(System.in);
    int n = in.nextInt();
    int x = in.nextInt();
    int[] a = new int[n];
    for (int i = 0; i < n; i++) {
        a[i] = in.nextInt();
    }
    int sum = 0;
    for (int i = 0; x != 0; x /= 2, i++) {
        if (x % 2 == 1) {
            sum += a[i];
        }
    }
    System.out.println(sum);
}

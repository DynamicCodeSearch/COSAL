public static void main(String[] args) {
    Scanner scan = new Scanner(System.in);
    int A = scan.nextInt();
    int B = scan.nextInt();
    if (A == B) {
        System.out.println("Draw");
    } else if (A == 1) {
        System.out.println("Alice");
    } else if (B == 1) {
        System.out.println("Bob");
    } else if (A > B) {
        System.out.println("Alice");
    } else {
        System.out.println("Bob");
    }
}

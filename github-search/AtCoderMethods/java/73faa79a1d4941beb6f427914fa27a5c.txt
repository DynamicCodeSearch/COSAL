
public static void main(String[] args) {
    
    Scanner sc = new Scanner(System.in);
    int N = sc.nextInt();
    int K = sc.nextInt();
    int[] x = new int[N];
    int move = 0;
    for (int i = 0; i < N; i++) {
        x[i] = sc.nextInt();
        if (x[i] > (K / 2)) {
            move += (K - x[i]) * 2;
        } else {
            move += x[i] * 2;
        }
    }
    System.out.println(move);
}

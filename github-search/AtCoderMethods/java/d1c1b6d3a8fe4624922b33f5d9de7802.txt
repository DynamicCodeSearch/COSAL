public static void main(String[] args) {
    Scanner scanner = new Scanner(System.in);
    String[][] s = new String[4][4];
    for (int i = 0; i < 4; i++) {
        for (int j = 0; j < 4; j++) {
            s[i][j] = scanner.next();
        }
    }
    for (int i = 3; i >= 0; i--) {
        System.out.printf("%s %s %s %s\n", s[i][3], s[i][2], s[i][1], s[i][0]);
    }
}

public static void main(String[] args) {
    Scanner sc = new Scanner(new InputStreamReader(System.in));
    String n = sc.next();
    if (n.equals("A")) {
        System.out.println("1");
    } else if (n.equals("B")) {
        System.out.println("2");
    } else if (n.equals("C")) {
        System.out.println("3");
    } else if (n.equals("D")) {
        System.out.println("4");
    } else {
        System.out.println("5");
    }
}

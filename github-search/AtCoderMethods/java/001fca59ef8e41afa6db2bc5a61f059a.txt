public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int a = sc.nextInt();
    int b = sc.nextInt();
    int c = sc.nextInt();
    int d = sc.nextInt();
    int startMax = Math.max(a, c);
    int stopMin = Math.min(b, d);
    if (startMax > stopMin) {
        System.out.println(0);
    } else {
        System.out.println(stopMin - startMax);
    }
}

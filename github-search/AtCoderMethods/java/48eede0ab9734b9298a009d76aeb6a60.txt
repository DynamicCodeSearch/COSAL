public static void main(String[] args) {
    Scanner sc = new Scanner(System.in);
    int a = sc.nextInt();
    int b = sc.nextInt();
    int c = sc.nextInt();
    int min = a + b;
    if (min > a + c) {
        min = a + c;
    }
    if (min > b + c) {
        min = b + c;
    }
    System.out.println(min);
}
